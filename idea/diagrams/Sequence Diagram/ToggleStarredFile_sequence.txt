@startuml
title Sequence Diagram - Toggle Starred File Use Case

actor Client
participant "FileController" as Controller
participant "ToggleStarredFileCommand" as Command
participant "ToggleStarredFileService" as Service
participant "FilePersistencePort" as FilePort

Client -> Controller: POST /api/v1/files/toggle/starred/{fileId}
activate Controller

Controller -> Command **: new(ownerId, fileId)
activate Command
Command --> Controller: command
deactivate Command

Controller -> Service: execute(command)
activate Service

Service -> Command: validate()
activate Command
Command --> Service
deactivate Command

Service -> FilePort: findByIdAndOwnerId(fileId, ownerId)
activate FilePort
FilePort --> Service: Optional<File> (targetFile)
deactivate FilePort

alt File not found
    Service -[#red]> Controller: throw DomainException
else File found
    Service -> FilePort: toggleStarred(fileId)
    activate FilePort
    FilePort --> Service: void
    deactivate FilePort
    
    Service --> Controller: ToggleStarredFileResult
end
deactivate Service

Controller --> Client: 200 OK\n(ToggleStarredFileResult)
deactivate Controller
@enduml
